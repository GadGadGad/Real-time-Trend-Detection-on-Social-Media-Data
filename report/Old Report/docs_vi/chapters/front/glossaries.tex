\addcontentsline{toc}{chapter}{Danh sách từ viết tắt}
\begin{tabular}{| p{.2\textwidth} |p{.7\textwidth} |}
        \hline
        \textbf{Từ viết tắt} & \textbf{Định nghĩa}                                                                                                               \\
        \hline
        LLM                  & Large Language Model (Mô hình Ngôn ngữ Lớn)                                                                                       \\
        \hline
        SAHC                 & Social-Aware Hierarchical Clustering (Phân cụm Phân cấp Nhận thức Mạng xã hội)                                                    \\
        \hline
        NMI                  & Normalized Mutual Information (Thông tin Tương hỗ Chuẩn hóa)                                                                      \\
        \hline
        NLP                  & Natural Language Processing (Xử lý Ngôn ngữ Tự nhiên)                                                                             \\
        \hline
        BERT                 & Bidirectional Encoder Representations from Transformers                                                                           \\
        \hline
        SBERT                & Sentence-BERT                                                                                                                     \\
        \hline
        LDA                  & Latent Dirichlet Allocation (Phân bổ Dirichlet Tiềm ẩn)                                                                           \\
        \hline
        TF-IDF               & Term Frequency-Inverse Document Frequency (Tần suất thuật ngữ - Nghịch đảo tần suất tài liệu)                                     \\
        \hline
        BM25                 & Best Matching 25                                                                                                                  \\
        \hline
        DBSCAN               & Density-Based Spatial Clustering of Applications with Noise (Phân cụm Không gian Dựa trên Mật độ với Nhiễu)                       \\
        \hline
        HDBScan              & Hierarchical Density-Based Spatial Clustering of Applications with Noise (Phân cụm Không gian Dựa trên Mật độ Phân cấp với Nhiễu) \\
        \hline
        RAG                  & Retrieval-Augmented Generation (Tạo sinh Tăng cường Truy xuất)                                                                    \\
        \hline
        API                  & Application Programming Interface (Giao diện Lập trình Ứng dụng)                                                                  \\
        \hline
        JSON                 & JavaScript Object Notation                                                                                                        \\
        \hline
        PII                  & Personally Identifiable Information (Thông tin Định danh Cá nhân)                                                                 \\
        \hline
